1. Get the HortusFox NodePort:
   export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "hortusfox.fullname" . }}-app)

2. Get your Node IP (this command varies by platform, e.g., for Minikube):
   export NODE_IP=$(minikube ip)

3. Access HortusFox at:
   http://$NODE_IP:$NODE_PORT
